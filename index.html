# pip install selenium from selenium import webdriver from selenium.webdriver.common.by import By import time driver = webdriver.Chrome() # 드라이버 준비(크롬드라이버는 동일폴더에) driver.get('https://studio.youtube.com/') # 유튜브 스튜디오 접속 input("유튜브 로그인(최초 1회만 직접) 후 엔터를 누르세요.") # 1회 로그인, 이후 자동 # 동영상 업로드 버튼 클릭 driver.find_element(By.XPATH, '//*[@id="create-icon"]').click() time.sleep(2) # '동영상 업로드' 메뉴 클릭 driver.find_element(By.XPATH, '//*[@id="text-item-0"]').click() time.sleep(2) # 파일 업로드 input 찾기 upload_input = driver.find_element(By.XPATH, '//input[@type="file"]') upload_input.send_keys(r'C:\video\테스트영상.mp4') # 업로드 파일 실제 경로 입력 # 제목/설명 자동 입력 (필요 시) time.sleep(8) driver.find_element(By.XPATH, '//*[@id="textbox"]').send_keys('자동화 테스트 영상') # (아래는 "다음" 3회 클릭 + 공개 버튼 클릭 등 추가 가능) # driver.find_element(By.XPATH, '//*[@id="next-button"]').click() # ... time.sleep(60) # 업로드 완료 대기 driver.quit()
