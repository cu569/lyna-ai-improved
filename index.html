# 1) 작업 폴더로 이동 / 새 폴더 생성 mkdir g911-rima && cd g911-rima # 2) .gitignore 생성 cat > .gitignore << 'EOF' __pycache__/ *.pyc chromedriver* .env EOF # 3) 의존성 목록 cat > requirements.txt << 'EOF' Flask selenium EOF # 4) Python 서버 + Selenium 자동업로드 (app.py) cat > app.py << 'EOF' from flask import Flask, render_template, jsonify from selenium import webdriver from selenium.webdriver.common.by import By import time, os, sys # ▶ 실제 업로드할 영상 파일 경로로 수정 (영문/공백 없는 경로 권장) VIDEO_PATH = r'C:\video\test.mp4' if not os.path.isfile(VIDEO_PATH): print(f"ERROR: 영상 파일이 없습니다: {VIDEO_PATH}") sys.exit(1) app = Flask(__name__) # 단일 HTML 템플릿 HTML = """ G911 리마 AI 상황판

G911 리마 AI 상황판

🎥

유튜브 업로드

실제 자동 업로드 실행
업로드 횟수: 0회 

  




"""

@app.route('/')
def index():
    return HTML

@app.route('/upload', methods=['POST'])
def upload():
    try:
        driver = webdriver.Chrome()  # chromedriver.exe는 PATH에!
        driver.get('https://studio.youtube.com/')
        input("▶ 로그인 완료 후 Enter:")  # 최초 1회만

        driver.find_element(By.XPATH, '//*[@id="create-icon"]').click()
        time.sleep(1)
        driver.find_element(By.XPATH, '//*[@id="text-item-0"]').click()
        time.sleep(2)
        inp = driver.find_element(By.XPATH, '//input[@type="file"]')
        inp.send_keys(VIDEO_PATH)
        time.sleep(60)
        driver.quit()
        return jsonify(ok=True, url="https://youtu.be/XXXXXXX")
    except Exception as e:
        return jsonify(ok=False, error=str(e))

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=8080)
EOF
