# app.py from flask import Flask, request, jsonify, render_template_string from selenium import webdriver from selenium.webdriver.common.by import By import time, os, sys # ▷ 실제 업로드할 영상 파일 경로로 수정하세요 (영문·공백 없는 경로 권장) VIDEO_PATH = r'C:\video\test.mp4' if not os.path.isfile(VIDEO_PATH): print(f"ERROR: 영상 파일이 없습니다: {VIDEO_PATH}") sys.exit(1) app = Flask(__name__) # ▷ 단일 HTML 템플릿 (index) HTML = """ G911 리마 AI 상황판 + 업로드

G911 리마 AI 상황판

🎥

유튜브 업로드

실제 자동 업로드 실행
업로드 횟수: 0회 

  




"""

@app.route('/')
def index():
    return render_template_string(HTML)

@app.route('/upload', methods=['POST'])
def upload():
    try:
        driver = webdriver.Chrome()  # PATH에 chromedriver.exe 필요
        driver.get('https://studio.youtube.com/')
        input("▶ 로그인 완료 후 Enter:")  # 최초 1회만 수동 로그인

        # 업로드 과정 자동화
        driver.find_element(By.XPATH, '//*[@id="create-icon"]').click()
        time.sleep(1)
        driver.find_element(By.XPATH, '//*[@id="text-item-0"]').click()
        time.sleep(2)
        inp = driver.find_element(By.XPATH, '//input[@type="file"]')
        inp.send_keys(VIDEO_PATH)
        time.sleep(60)  # 업로드 완료 대기
        driver.quit()

        # 실제 URL 스크래핑 추가 가능
        return jsonify(ok=True, url="https://youtu.be/XXXXXXX")
    except Exception as e:
        return jsonify(ok=False, error=str(e))

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=8080)
